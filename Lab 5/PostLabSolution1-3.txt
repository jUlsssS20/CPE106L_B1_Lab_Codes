CREATE TABLE PARTICIPANT (
    PARTICIPANT_NUM INTEGER PRIMARY KEY,
    LAST_NAME TEXT NOT NULL,
    FIRST_NAME TEXT NOT NULL,
    ADDRESS TEXT NOT NULL,
    CITY TEXT NOT NULL,
    STATE TEXT NOT NULL,
    POSTAL_CODE TEXT NOT NULL,
    TELEPHONE_NUMBER TEXT NOT NULL,
    DATE_OF_BIRTH DATE NOT NULL
);

CREATE TABLE ADVENTURE_CLASS (
    CLASS_NUM INTEGER PRIMARY KEY,
    CLASS_DESCRIPTION TEXT NOT NULL,
    MAX_PEOPLE INTEGER NOT NULL,
    CLASS_FEE REAL NOT NULL
);

CREATE TABLE ENROLLMENT (
    PARTICIPANT_NUM INTEGER NOT NULL,
    CLASS_NUM INTEGER NOT NULL,
    ENROLLMENT_DATE DATE NOT NULL,
    PRIMARY KEY (PARTICIPANT_NUM, CLASS_NUM),
    FOREIGN KEY (PARTICIPANT_NUM) REFERENCES PARTICIPANT (PARTICIPANT_NUM),
    FOREIGN KEY (CLASS_NUM) REFERENCES ADVENTURE_CLASS (CLASS_NUM)
);

SELECT
    P.PHONE,
    P.LAST_NAME,
    P.FIRST_NAME,
    C.CLASS_DESCRIPTION,
    E.ENROLLMENT_DATE
FROM
    PARTICIPANT P
JOIN
    ENROLLMENT E ON P.PARTICIPANT_NUM = E.PARTICIPANT_NUM
JOIN
    ADVENTURE_CLASS C ON E.CLASS_NUM = C.CLASS_NUM;

SELECT
    C.CLASS_DESCRIPTION,
    P.PHONE,
    P.LAST_NAME,
    P.FIRST_NAME,
    E.ENROLLMENT_DATE
FROM
    ADVENTURE_CLASS C
JOIN
    ENROLLMENT E ON C.CLASS_NUM = E.CLASS_NUM
JOIN
    PARTICIPANT P ON E.PARTICIPANT_NUM = P.PARTICIPANT_NUM;

CREATE DATABASE solmaris_condominium;

USE solmaris_condominium;

CREATE TABLE renters (
    renter_number INT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    middle_initial VARCHAR(1) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    address VARCHAR(100) NOT NULL,
    city VARCHAR(50) NOT NULL,
    state VARCHAR(2) NOT NULL,
    postal_code VARCHAR(5) NOT NULL,
    telephone_number VARCHAR(12) NOT NULL,
    email_address VARCHAR(50) NOT NULL
);

CREATE TABLE properties (
    condo_location_number INT PRIMARY KEY,
    condo_location_name VARCHAR(50) NOT NULL,
    address VARCHAR(100) NOT NULL,
    city VARCHAR(50) NOT NULL,
    state VARCHAR(2) NOT NULL,
    postal_code VARCHAR(5) NOT NULL,
    condo_unit_number INT NOT NULL,
    square_footage INT NOT NULL,
    number_of_bedrooms INT NOT NULL,
    number_of_bathrooms INT NOT NULL,
    maximum_number_of_persons INT NOT NULL,
    base_weekly_rate DECIMAL(10, 2) NOT NULL
);

CREATE TABLE rental_agreements (
    rental_agreement_number INT PRIMARY KEY,
    renter_number INT NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    middle_initial VARCHAR(1) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    address VARCHAR(100) NOT NULL,
    city VARCHAR(50) NOT NULL,
    state VARCHAR(2) NOT NULL,
    postal_code VARCHAR(5) NOT NULL,
    telephone_number VARCHAR(12) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    weekly_rental_amount DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (renter_number) REFERENCES renters(renter_number)
);
